package com.bit.coin.p2p.quic.control;

import com.bit.coin.p2p.quic.QuicConstants;
import com.bit.coin.p2p.quic.QuicFrame;
import com.bit.coin.p2p.quic.SendQuicData;
import lombok.extern.slf4j.Slf4j;

import java.net.InetSocketAddress;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.*;
import java.util.concurrent.atomic.AtomicBoolean;
import java.util.concurrent.atomic.AtomicInteger;

@Slf4j
public class GlobalDataTransmissionController {

    // 帧发送任务队列（优先级：FIFO，线程安全）
    private final BlockingQueue<QuicFrame> frameSendQueue;

    // 数据发送任务队列（整段数据，会被分片为帧后加入帧队列）
    private final BlockingQueue<DataSendTask> dataSendQueue;

    // 数据重试队列（存储入队失败的任务，等待重试）
    private final BlockingQueue<RetryableDataTask> retryDataQueue;

    // 发送控制核心参数
    private final int sendRate; // 每秒发送的最大帧数（控制速率核心）
    private final long sendIntervalMs; // 每帧发送间隔（ms），由sendRate计算得出

    // 重试相关参数
    private static final int MAX_RETRY_TIMES = 5; // 最大重试次数
    private static final long BASE_RETRY_DELAY_MS = 100; // 基础重试延迟（ms）

    // 发送线程相关
    private final ScheduledExecutorService sendExecutor;
    private final AtomicBoolean isRunning;
    private final AtomicInteger currentQueueSize; // 当前帧队列大小监控

    // 单例实例
    private static volatile GlobalDataTransmissionController INSTANCE;

    /**
     * 数据发送任务封装（整段数据+连接信息）
     */
    @lombok.Data
    private static class DataSendTask {
        private long connectionId;
        private long dataId;
        private byte[] data;
        private InetSocketAddress remoteAddress;
        private int maxFramePayload; // 单帧最大载荷
    }

    /**
     * 可重试的数据任务封装（包含重试次数）
     */
    @lombok.Data
    private static class RetryableDataTask {
        private DataSendTask dataTask;
        private int retryCount; // 当前重试次数
        private long lastRetryTime; // 上次重试时间戳（ms）
    }

    /**
     * 私有构造器（单例模式）
     * @param sendRate 每秒发送帧数（建议值：1000-5000，根据网络环境调整）
     * @param queueCapacity 队列容量（防止内存溢出）
     */
    private GlobalDataTransmissionController(int sendRate, int queueCapacity) {
        // 初始化队列（使用LinkedBlockingQueue保证FIFO和线程安全）
        this.frameSendQueue = new LinkedBlockingQueue<>(queueCapacity);
        this.dataSendQueue = new LinkedBlockingQueue<>(queueCapacity);
        this.retryDataQueue = new LinkedBlockingQueue<>(queueCapacity); // 重试队列容量与原队列一致

        // 速率控制核心参数
        this.sendRate = Math.max(1, sendRate); // 最小1帧/秒
        this.sendIntervalMs = 1000 / this.sendRate; // 计算每帧发送间隔

        // 初始化发送线程池（核心线程+重试线程，保证发送顺序和速率稳定）
        this.sendExecutor = new ScheduledThreadPoolExecutor(2, runnable -> {
            Thread thread = new Thread(runnable, "quic-frame-sender-thread");
            thread.setDaemon(true); // 守护线程，不阻塞程序退出
            thread.setPriority(Thread.NORM_PRIORITY);
            return thread;
        });

        this.isRunning = new AtomicBoolean(false);
        this.currentQueueSize = new AtomicInteger(0);

        // 启动发送任务
        startSender();
    }

    /**
     * 获取单例实例（默认配置：2000帧/秒，队列容量10万）
     */
    public static GlobalDataTransmissionController getInstance() {
        if (INSTANCE == null) {
            synchronized (GlobalDataTransmissionController.class) {
                if (INSTANCE == null) {
                    // 默认配置：2000帧/秒，队列容量100000
                    INSTANCE = new GlobalDataTransmissionController(2000, 100000);
                }
            }
        }
        return INSTANCE;
    }

    /**
     * 启动帧发送任务（核心：按固定间隔从队列取帧发送）
     */
    private void startSender() {
        if (isRunning.compareAndSet(false, true)) {
            // 启动数据分片任务（处理原始数据队列）
            sendExecutor.submit(this::processDataQueue);
            // 启动重试任务处理线程（处理入队失败的任务）
            sendExecutor.submit(this::processRetryDataQueue);

            // 启动帧发送任务（按固定间隔发送）
            sendExecutor.scheduleAtFixedRate(this::sendFrameFromQueue,
                    0, // 初始延迟0ms
                    sendIntervalMs, // 发送间隔（核心速率控制）
                    TimeUnit.MILLISECONDS);

            log.info("QUIC帧发送控制器已启动 | 发送速率：{}帧/秒 | 发送间隔：{}ms | 队列容量：{}",
                    sendRate, sendIntervalMs, frameSendQueue.remainingCapacity() + frameSendQueue.size());
        }
    }

    /**
     * 处理原始数据队列：将整段数据分片为帧，保证原子性入队
     */
    private void processDataQueue() {
        while (isRunning.get()) {
            try {
                // 阻塞获取数据任务（队列为空时等待）
                DataSendTask dataTask = dataSendQueue.take();
                processSingleDataTask(dataTask, 0); // 初始重试次数为0
            } catch (InterruptedException e) {
                log.warn("数据队列处理线程被中断，准备退出");
                Thread.currentThread().interrupt();
                break;
            } catch (Exception e) {
                log.error("数据队列处理异常", e);
            }
        }
    }

    /**
     * 处理重试数据队列：延迟重试入队失败的任务
     */
    private void processRetryDataQueue() {
        while (isRunning.get()) {
            try {
                // 阻塞获取重试任务
                RetryableDataTask retryTask = retryDataQueue.take();
                // 检查是否达到最大重试次数
                if (retryTask.getRetryCount() >= MAX_RETRY_TIMES) {
                    log.error("数据任务重试次数达到上限，放弃发送 | 连接ID：{} | 数据ID：{} | 重试次数：{}",
                            retryTask.getDataTask().getConnectionId(),
                            retryTask.getDataTask().getDataId(),
                            retryTask.getRetryCount());
                    continue;
                }

                // 计算退避延迟（指数退避：100ms, 200ms, 400ms, 800ms, 1600ms...）
                long delayMs = BASE_RETRY_DELAY_MS * (1 << retryTask.getRetryCount());
                long currentTime = System.currentTimeMillis();
                long waitTime = retryTask.getLastRetryTime() + delayMs - currentTime;

                // 等待退避时间（避免频繁重试）
                if (waitTime > 0) {
                    Thread.sleep(waitTime);
                }

                // 重新处理该任务
                processSingleDataTask(retryTask.getDataTask(), retryTask.getRetryCount() + 1);
            } catch (InterruptedException e) {
                log.warn("重试数据队列处理线程被中断，准备退出");
                Thread.currentThread().interrupt();
                break;
            } catch (Exception e) {
                log.error("重试数据队列处理异常", e);
            }
        }
    }

    /**
     * 处理单个数据任务（核心：保证帧原子性入队）
     * @param dataTask 数据任务
     * @param currentRetryCount 当前重试次数
     */
    private void processSingleDataTask(DataSendTask dataTask, int currentRetryCount) {
        List<QuicFrame> tempFrameList = new ArrayList<>();
        boolean allEnqueueSuccess = true;

        try {
            // 1. 分片数据为帧，暂存到临时列表
            SendQuicData sendQuicData = SendQuicData.buildFromFullData(
                    dataTask.getConnectionId(),
                    dataTask.getDataId(),
                    dataTask.getData(),
                    dataTask.getRemoteAddress(),
                    dataTask.getMaxFramePayload()
            );

            QuicFrame[] frameArray = sendQuicData.getFrameArray();
            if (frameArray == null || frameArray.length == 0) {
                log.warn("数据分片后无帧生成 | 连接ID：{} | 数据ID：{}",
                        dataTask.getConnectionId(), dataTask.getDataId());
                return;
            }

            // 收集所有非空帧到临时列表
            for (QuicFrame frame : frameArray) {
                if (frame != null) {
                    tempFrameList.add(frame);
                }
            }

            if (tempFrameList.isEmpty()) {
                log.warn("数据分片后无有效帧 | 连接ID：{} | 数据ID：{}",
                        dataTask.getConnectionId(), dataTask.getDataId());
                return;
            }

            // 2. 尝试原子性入队所有帧
            List<QuicFrame> enqueuedFrames = new ArrayList<>();
            try {
                for (QuicFrame frame : tempFrameList) {
                    // 尝试入队（阻塞10ms，保证有机会等待队列空间）
                    boolean offerSuccess = frameSendQueue.offer(frame, 10, TimeUnit.MILLISECONDS);
                    if (offerSuccess) {
                        enqueuedFrames.add(frame);
                        currentQueueSize.incrementAndGet();
                    } else {
                        allEnqueueSuccess = false;
                        break; // 任一帧入队失败，终止后续入队
                    }
                }
            } catch (InterruptedException e) {
                allEnqueueSuccess = false;
                Thread.currentThread().interrupt();
                log.warn("帧入队过程被中断 | 连接ID：{} | 数据ID：{}",
                        dataTask.getConnectionId(), dataTask.getDataId());
            }

            // 3. 处理入队结果
            if (allEnqueueSuccess) {
                // 全部入队成功
                log.debug("数据分片完成并原子性入队 | 连接ID：{} | 数据ID：{} | 生成帧数：{} | 重试次数：{}",
                        dataTask.getConnectionId(), dataTask.getDataId(), tempFrameList.size(), currentRetryCount);
            } else {
                // 部分入队失败：回滚已入队的帧，释放所有帧资源
                for (QuicFrame frame : enqueuedFrames) {
                    frameSendQueue.remove(frame); // 移除已入队的帧
                    currentQueueSize.decrementAndGet();
                    frame.release(); // 释放资源
                }
                for (QuicFrame frame : tempFrameList) {
                    if (!enqueuedFrames.contains(frame)) {
                        frame.release(); // 释放未入队的帧资源
                    }
                }

                // 将任务加入重试队列
                RetryableDataTask retryTask = new RetryableDataTask();
                retryTask.setDataTask(dataTask);
                retryTask.setRetryCount(currentRetryCount);
                retryTask.setLastRetryTime(System.currentTimeMillis());

                boolean retryOfferSuccess = retryDataQueue.offer(retryTask);
                if (retryOfferSuccess) {
                    log.warn("数据帧入队失败，加入重试队列 | 连接ID：{} | 数据ID：{} | 重试次数：{} | 最大重试次数：{}",
                            dataTask.getConnectionId(), dataTask.getDataId(), currentRetryCount, MAX_RETRY_TIMES);
                } else {
                    log.error("重试队列已满，无法加入重试任务 | 连接ID：{} | 数据ID：{} | 重试次数：{}",
                            dataTask.getConnectionId(), dataTask.getDataId(), currentRetryCount);
                }
            }
        } catch (Exception e) {
            // 分片/入队过程异常：释放所有暂存的帧资源
            for (QuicFrame frame : tempFrameList) {
                try {
                    frame.release();
                } catch (Exception ex) {
                    log.error("释放帧资源失败", ex);
                }
            }
            log.error("处理数据任务异常 | 连接ID：{} | 数据ID：{} | 重试次数：{}",
                    dataTask.getConnectionId(), dataTask.getDataId(), currentRetryCount, e);

            // 异常时也加入重试队列（除非达到最大重试次数）
            if (currentRetryCount < MAX_RETRY_TIMES) {
                RetryableDataTask retryTask = new RetryableDataTask();
                retryTask.setDataTask(dataTask);
                retryTask.setRetryCount(currentRetryCount);
                retryTask.setLastRetryTime(System.currentTimeMillis());
                retryDataQueue.offer(retryTask);
            }
        }
    }

    /**
     * 从帧队列取帧并发送（核心速率控制逻辑）
     */
    private void sendFrameFromQueue() {
        if (!isRunning.get()) {
            return;
        }

        // 非阻塞获取帧（避免线程阻塞，保证速率稳定）
        QuicFrame frame = frameSendQueue.poll();
        if (frame != null) {
            try {
                // 发送帧（复用现有发送逻辑）
                sendSingleFrame(frame);
                currentQueueSize.decrementAndGet();
            } catch (Exception e) {
                log.error("帧发送失败 | 连接ID：{} | 数据ID：{} | 序列号：{}",
                        frame.getConnectionId(), frame.getDataId(), frame.getSequence(), e);
                frame.release(); // 发送失败也要释放资源
            }
        } else {
            // 队列为空时的监控日志（可选）
            if (currentQueueSize.get() == 0 && log.isDebugEnabled()) {
                log.debug("帧发送队列为空，暂无数据发送");
            }
        }

        // 定期打印队列监控信息（每10秒打印一次）
        if (System.currentTimeMillis() % 10000 < sendIntervalMs) {
            log.info("帧发送队列监控 | 当前队列大小：{} | 剩余容量：{} | 累计发送速率：{}帧/秒 | 重试队列大小：{}",
                    currentQueueSize.get(),
                    frameSendQueue.remainingCapacity(),
                    sendRate,
                    retryDataQueue.size());
        }
    }

    /**
     * 发送单个帧（复用现有Netty发送逻辑）
     */
    private void sendSingleFrame(QuicFrame frame) {
        if (frame == null || frame.getRemoteAddress() == null) {
            log.warn("无效帧，跳过发送 | 帧：{}", frame);
            return;
        }

        try {
            // 编码帧为ByteBuf
            io.netty.buffer.ByteBuf buf = QuicConstants.ALLOCATOR.buffer();
            frame.encode(buf);

            // 构建UDP数据包并发送
            io.netty.channel.socket.DatagramPacket packet = new io.netty.channel.socket.DatagramPacket(
                    buf, frame.getRemoteAddress()
            );

            // 使用全局Channel发送（与现有逻辑保持一致）
            if (com.bit.coin.p2p.quic.QuicConnectionManager.Global_Channel != null) {
                com.bit.coin.p2p.quic.QuicConnectionManager.Global_Channel.writeAndFlush(packet)
                        .addListener(future -> {
                            if (!future.isSuccess()) {
                                log.error("帧发送失败（网络层） | 连接ID：{} | 数据ID：{} | 序列号：{}",
                                        frame.getConnectionId(), frame.getDataId(), frame.getSequence(), future.cause());
                            } else {
                                log.debug("帧发送成功 | 连接ID：{} | 数据ID：{} | 序列号：{}",
                                        frame.getConnectionId(), frame.getDataId(), frame.getSequence());
                            }
                            // 无论成功失败，都释放帧资源
                            frame.release();
                        });
            } else {
                log.error("全局Channel为空，帧发送失败 | 连接ID：{} | 数据ID：{} | 序列号：{}",
                        frame.getConnectionId(), frame.getDataId(), frame.getSequence());
                frame.release();
            }
        } catch (Exception e) {
            log.error("帧编码/发送异常", e);
            frame.release();
        }
    }

    // -------------------------- 对外提供的核心接口 --------------------------

    /**
     * 提交整段数据到发送队列（会自动分片为帧）
     * @param connectionId 连接ID
     * @param dataId 数据ID
     * @param data 待发送的原始数据
     * @param remoteAddress 目标地址
     * @param maxFramePayload 单帧最大载荷
     * @return 是否提交成功（队列满时返回false）
     */
    public boolean submitData(long connectionId, long dataId, byte[] data,
                              InetSocketAddress remoteAddress, int maxFramePayload) {
        if (!isRunning.get()) {
            log.error("帧发送控制器未运行，数据提交失败");
            return false;
        }
        if (data == null || data.length == 0 || remoteAddress == null) {
            log.warn("无效数据，提交失败 | 连接ID：{} | 数据ID：{}", connectionId, dataId);
            return false;
        }

        DataSendTask task = new DataSendTask();
        task.setConnectionId(connectionId);
        task.setDataId(dataId);
        task.setData(data);
        task.setRemoteAddress(remoteAddress);
        task.setMaxFramePayload(maxFramePayload);

        // 非阻塞提交（队列满时返回false）
        boolean success = dataSendQueue.offer(task);
        if (success) {
            log.debug("数据提交成功 | 连接ID：{} | 数据ID：{} | 数据长度：{} | 当前数据队列大小：{}",
                    connectionId, dataId, data.length, dataSendQueue.size());
        } else {
            log.warn("数据队列已满，提交失败 | 连接ID：{} | 数据ID：{} | 数据长度：{}",
                    connectionId, dataId, data.length);
        }
        return success;
    }

    /**
     * 直接提交单个帧到发送队列
     * @param frame 待发送的帧
     * @return 是否提交成功（队列满时返回false）
     */
    public boolean submitFrame(QuicFrame frame) {
        if (!isRunning.get()) {
            log.error("帧发送控制器未运行，帧提交失败");
            return false;
        }
        if (frame == null) {
            log.warn("无效帧，提交失败");
            return false;
        }

        // 非阻塞提交（队列满时返回false）
        boolean success = frameSendQueue.offer(frame);
        if (success) {
            currentQueueSize.incrementAndGet();
            log.debug("帧提交成功 | 连接ID：{} | 数据ID：{} | 序列号：{} | 当前帧队列大小：{}",
                    frame.getConnectionId(), frame.getDataId(), frame.getSequence(), frameSendQueue.size());
        } else {
            log.warn("帧队列已满，提交失败 | 连接ID：{} | 数据ID：{} | 序列号：{}",
                    frame.getConnectionId(), frame.getDataId(), frame.getSequence());
            frame.release(); // 提交失败释放帧资源
        }
        return success;
    }

    // -------------------------- 辅助方法 --------------------------

    /**
     * 获取当前帧队列大小
     */
    public int getFrameQueueSize() {
        return frameSendQueue.size();
    }

    /**
     * 获取当前数据队列大小
     */
    public int getDataQueueSize() {
        return dataSendQueue.size();
    }

    /**
     * 获取当前重试队列大小
     */
    public int getRetryQueueSize() {
        return retryDataQueue.size();
    }

    /**
     * 停止发送控制器（优雅关闭）
     */
    public void stop() {
        if (isRunning.compareAndSet(true, false)) {
            // 关闭线程池
            sendExecutor.shutdown();
            try {
                // 等待线程池终止（最多等待10秒）
                if (!sendExecutor.awaitTermination(10, TimeUnit.SECONDS)) {
                    sendExecutor.shutdownNow();
                }
            } catch (InterruptedException e) {
                sendExecutor.shutdownNow();
                Thread.currentThread().interrupt();
            }

            // 清空所有队列并释放帧资源
            clearQueue(frameSendQueue);
            clearQueue(dataSendQueue);
            clearQueue(retryDataQueue);

            log.info("QUIC帧发送控制器已停止 | 已清空队列 | 释放资源完成");
        }
    }

    /**
     * 清空队列并释放资源
     */
    private <T> void clearQueue(BlockingQueue<T> queue) {
        T element;
        while ((element = queue.poll()) != null) {
            if (element instanceof QuicFrame) {
                ((QuicFrame) element).release();
            }
            // 其他类型资源可在此扩展
        }
    }
}
